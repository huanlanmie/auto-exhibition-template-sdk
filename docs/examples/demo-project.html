完整项目示例

一个覆盖多种字段类型的完整模板项目示例，从 configJson 定义到页面组件取值的完整链路。

项目结构

src/
├── template/
│   └── config.ts          // 模板配置
├── components/
│   ├── HeroSection.vue    // 主视觉区域
│   ├── Timeline.vue       // 时间轴组件
│   └── Downloads.vue      // 下载区域
├── App.vue
├── main.ts
└── vite.config.ts

configJson 配置

// src/template/config.ts
import { defineTemplateConfig } from 'auto-exhibition-template-sdk/config'

const configJson = defineTemplateConfig({
  meta: {
    name: '星图态势科技舱',
    code: 'starmap-tech-cabin',
    version: '1.0.0'
  },
  dataSchema: {
    fields: [
      // 主视觉区域
      { key: 'title', type: 'string', label: '主标题', value: '星图态势科技舱' },
      { key: 'subtitle', type: 'string', label: '副标题', value: '实时态势感知平台' },
      { key: 'poster', type: 'image', label: '主视觉海报', value: '/assets/template/poster-core.svg' },
      { key: 'showBadge', type: 'boolean', label: '显示徽标', value: true },

      // 时间轴
      {
        key: 'timeline',
        type: 'array',
        label: '时间轴',
        operations: [],
        value: [
          { key: 'phase01', type: 'string', label: '阶段一', value: '接入边缘传感节点。' },
          { key: 'phase02', type: 'string', label: '阶段二', value: '将采集数据转换为状态向量。' },
          { key: 'phase03', type: 'string', label: '阶段三', value: '完成全域态势融合推演。' }
        ]
      },

      // 下载资源
      {
        key: 'guideFile',
        type: 'file',
        label: '讲解文档',
        value: '/assets/template/guide.pdf',
        accept: ['image', '.pdf']
      }
    ]
  },
  functions: {}
})

export default configJson

vite.config.ts

import { defineConfig } from 'vite'
import vue from '@vitejs/plugin-vue'
import configJson from './src/template/config'
import { templateSdkPlugin } from 'auto-exhibition-template-sdk/vite'

export default defineConfig({
  plugins: [
    templateSdkPlugin({ configJson }),
    vue()
  ]
})

main.ts

import { createApp } from 'vue'
import App from './App.vue'
import TemplateSdk from 'auto-exhibition-template-sdk'

const app = createApp(App)
app.use(TemplateSdk)
app.mount('#app')

HeroSection.vue

<script setup>
import { useTemplateValue } from 'auto-exhibition-template-sdk'

const title = useTemplateValue('title', '')
const subtitle = useTemplateValue('subtitle', '')
const poster = useTemplateValue('poster', { url: '' })
const showBadge = useTemplateValue('showBadge', false)
</script>

<template>
  <section class="hero">
    <img :src="poster?.url" class="hero-poster" alt="" />
    <span v-if="showBadge" class="badge">NEW</span>
    <h1>{{ title }}</h1>
    <p>{{ subtitle }}</p>
  </section>
</template>

Timeline.vue

<script setup>
import { useTemplateValue } from 'auto-exhibition-template-sdk'

const timeline = useTemplateValue('timeline', [])
</script>

<template>
  <ol class="timeline">
    <li v-for="item in timeline" :key="item.key">
      {{ item.value }}
    </li>
  </ol>
</template>

Downloads.vue

<script setup>
import { useTemplateValue } from 'auto-exhibition-template-sdk'

const guideFile = useTemplateValue('guideFile', '')
</script>

<template>
  <div class="downloads">
    <a v-if="guideFile" :href="guideFile" target="_blank">
      下载讲解文档
    </a>
  </div>
</template>
